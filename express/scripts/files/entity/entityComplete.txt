import { Column, DataType, Table,Model } from "sequelize-typescript";

@Table({ tableName: "${props}" })
export class ${Props}Entity  extends Model implement ${Props}EntitySchema{
    @Column({
        allowNull: false,
        unique: true,
        type: DataType.STRING,
    })
    name: string;

    @Column({
        allowNull: false,
        unique: true,
        type: DataType.STRING,
    })
    slug: string;
}
